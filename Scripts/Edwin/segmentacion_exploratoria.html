Etapa 3 — Segmentacion exploratoria (Seccion 5.3.9)

Dataset \(D_{\mathrm{mod}}\): data_agrupada_mod.csv (cuadrante × semana, 2024–2025).

Profundiza la segmentacion territorial introducida en la Seccion 5.3.6: perfiles por nivel de actividad, comparacion temporal (2024 vs 2025), regimenes climaticos semanales y clustering multivariado de cuadrantes. Objetivo: evaluar heterogeneidad territorial y orientar estratificacion en la Etapa 4.

Metodologia aplicada:

  1. Perfiles por nivel de actividad (Baja / Media / Alta).
  2. Comparacion 2024 vs 2025 por estrato territorial.
  3. Regimen climatico semanal (seco vs lluvioso).
  4. Clustering k-means de cuadrantes (perfil operativo-climatico).
  5. Mapa y heatmap de perfiles por segmento.

cuadrante_resumen <- d_mod %>%
  group_by(id_cuadrante, latitud, longitud, altitud) %>%
  summarise(
    n_eventos = sum(n_eventos, na.rm = TRUE),
    usuarios = sum(usuarios_afectados, na.rm = TRUE),
    duracion_h = sum(duracion_total, na.rm = TRUE),
    semanas_activas = sum(n_eventos > 0, na.rm = TRUE),
    precip_media = mean(precip_total_semana, na.rm = TRUE),
    temp_media = mean(temp_media_semana, na.rm = TRUE),
    viento_medio = mean(viento_medio_semana, na.rm = TRUE),
    .groups = "drop"
  )

q_pos <- cuadrante_resumen$n_eventos[cuadrante_resumen$n_eventos > 0]

cuadrante_resumen <- cuadrante_resumen %>%
  mutate(
    nivel = case_when(
      n_eventos == 0 ~ "Sin eventos",
      n_eventos <= quantile(q_pos, 0.50) ~ "Baja",
      n_eventos <= quantile(q_pos, 0.85) ~ "Media",
      TRUE ~ "Alta"
    ),
    nivel = factor(nivel, levels = c("Sin eventos", "Baja", "Media", "Alta"))
  )

tabla_niveles <- cuadrante_resumen %>%

umbral_lluvia <- quantile(d_mod$precip_total_semana, 0.75, na.rm = TRUE)

d_mod <- d_mod %>%

vars_cluster <- c(
  "n_eventos", "usuarios", "duracion_h", "semanas_activas",

cuadrantes_cluster <- cuadrante_resumen %>%
  filter(n_eventos > 0) %>%
  select(id_cuadrante, latitud, longitud, all_of(vars_cluster)) %>%
  drop_na(all_of(vars_cluster))

mat_raw <- cuadrantes_cluster %>%
  select(all_of(vars_cluster))

sd_cluster <- sapply(mat_raw, sd, na.rm = TRUE)
vars_cluster_ok <- names(sd_cluster)[sd_cluster > 0 & !is.na(sd_cluster)]

mat_cluster <- scale(mat_raw[, vars_cluster_ok, drop = FALSE])
filas_ok <- complete.cases(mat_cluster)
mat_cluster <- mat_cluster[filas_ok, , drop = FALSE]
cuadrantes_cluster <- cuadrantes_cluster[filas_ok, ]

set.seed(42)
k_optimo <- min(4L, nrow(mat_cluster))
km <- kmeans(mat_cluster, centers = k_optimo, nstart = 25)

cuadrantes_cluster <- cuadrantes_cluster %>%
  mutate(cluster = factor(km$cluster, labels = paste0("C", seq_len(k_optimo))))

perfil_cluster <- cuadrantes_cluster %>%

  summarise(
    n_cuadrantes = n(),
