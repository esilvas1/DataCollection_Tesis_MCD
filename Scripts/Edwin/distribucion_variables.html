guardar_plot <- function(plot, archivo, width, height) {
  ggsave(
    file.path(UBICACION_DATA, archivo),
    plot = plot,
    width = width,
    height = height,
    dpi = 300,
    bg = "white"
  )
  if (dir.exists(path_img_tesis)) {
    file.copy(
      file.path(UBICACION_DATA, archivo),
      file.path(path_img_tesis, archivo),
      overwrite = TRUE
    )
  }
}

Indicadores de forma por variable

Asimetría, curtosis y proporción de ceros: soporte cuantitativo para interpretar las formas observadas en los diagramas de densidad.

asimetria <- function(x) {
  x <- x[!is.na(x)]
  m <- mean(x); s <- sd(x)
  mean((x - m)^3) / s^3
}

curtosis <- function(x) {
  x <- x[!is.na(x)]
  m <- mean(x); s <- sd(x)
  mean((x - m)^4) / s^4
}

n_modas <- function(x) {
  x <- x[!is.na(x)]
  d <- density(x)
  # picos locales de la densidad estimada
  sum(diff(sign(diff(d$y))) == -2)
}

tabla_distribucion <- map_dfr(vars_evaluar, function(v) {
  x <- tabla_base_agrupada[[v]]
  tibble(
    variable   = v,
    grupo      = ifelse(v %in% vars_clima, "clima", "evento"),
    pct_ceros  = round(100 * mean(x == 0, na.rm = TRUE), 2),
    asimetria  = round(asimetria(x), 3),
    curtosis   = round(curtosis(x), 3),
    n_modas    = n_modas(x)
  )
})

Distribución de variables climáticas

Diagramas de densidad para evaluar asimetría, dispersión y unimodalidad. Las temperaturas y la humedad se esperan estables y aproximadamente unimodales.

densidad_clima <- tabla_base_agrupada %>%
  select(all_of(setdiff(vars_clima, "precip_total_semana"))) %>%
  pivot_longer(everything(), names_to = "variable", values_to = "valor") %>%
  filter(!is.na(valor)) %>%
  ggplot(aes(x = valor)) +
  geom_density(fill = "#90CAF9", alpha = 0.65, color = "#1565C0") +
  facet_wrap(~ variable, scales = "free", ncol = 3) +
  labs(
    title = "Distribucion de variables climaticas semanales (sin precipitacion)",
    subtitle = paste0("Data agrupada cuadrante x semana | N = ", format(N_panel, big.mark = ",")),
    x = NULL, y = "Densidad"
  ) +
  theme_minimal(base_size = 11) +
  theme(plot.title = element_text(face = "bold"))

Precipitación acumulada: asimetría positiva

La precipitación semanal concentra muchas semanas con valores bajos o nulos y pocas observaciones extremas. Se muestra en escala original y en escala log(1 + x).

precip_larga <- tabla_base_agrupada %>%
  transmute(
    `Escala original (mm)`  = precip_total_semana,
    `Escala log(1 + x)`     = log1p(precip_total_semana)

  pivot_longer(everything(), names_to = "escala", values_to = "valor")

Distribución de variables de severidad y continuidad

Concentración en cero

La mayor parte de las semanas-cuadrante no registra fallas, por lo que estas variables están altamente sesgadas hacia cero.

tabla_ceros <- tabla_distribucion %>%
  filter(grupo == "evento") %>%
  mutate(variable = fct_reorder(variable, pct_ceros, .desc = TRUE))

grafico_ceros <- ggplot(tabla_ceros, aes(x = variable, y = pct_ceros, fill = variable)) +
  geom_col(width = 0.65, show.legend = FALSE) +

  scale_fill_manual(values = c("#EF9A9A", "#FFCC80", "#A5D6A7", "#90CAF9")) +

Síntesis

tabla_distribucion %>%
  mutate(
    lectura = case_when(
      grupo == "clima" & abs(asimetria) <= 0.5 ~ "estable, aprox. unimodal/simetrica",
      grupo == "clima" & asimetria > 0.5       ~ "asimetria positiva marcada",
      grupo == "evento"                        ~ "sesgada hacia cero, cola derecha larga",
      TRUE                                     ~ "revisar"
    )
  ) %>%
  select(variable, grupo, pct_ceros, asimetria, curtosis, lectura)
