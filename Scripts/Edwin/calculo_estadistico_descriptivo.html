Medidas de tendencia central

Media, mediana y moda. Para variables discretas (conteos) la moda es el valor más frecuente; para variables continuas se estima como el punto de máxima densidad.

moda_discreta <- function(x) {
  x <- x[!is.na(x)]
  ux <- unique(x)
  ux[which.max(tabulate(match(x, ux)))]
}

moda_continua <- function(x) {
  x <- x[!is.na(x)]
  d <- density(x)
  d$x[which.max(d$y)]
}

tabla_tendencia_central <- map_dfr(vars_evaluar, function(v) {
  x <- tabla_base_agrupada[[v]]
  es_discreta <- all(x[!is.na(x)] == round(x[!is.na(x)]))
  tibble(
    variable = v,
    media    = mean(x, na.rm = TRUE),
    mediana  = median(x, na.rm = TRUE),
    moda     = if (es_discreta) moda_discreta(x) else moda_continua(x),
    tipo_moda = if (es_discreta) "valor mas frecuente" else "pico de densidad"
  )
}) %>%
  mutate(across(where(is.numeric), ~ round(., 3)))

Medidas de dispersión

Rango, varianza, desviación estándar y coeficiente de variación (CV = s / media).

tabla_dispersion <- map_dfr(vars_evaluar, function(v) {
  x <- tabla_base_agrupada[[v]]
  x <- x[!is.na(x)]
  tibble(
    variable  = v,
    minimo    = min(x),
    maximo    = max(x),
    rango     = max(x) - min(x),
    varianza  = var(x),
    desv_est  = sd(x),
    coef_var  = ifelse(mean(x) != 0, sd(x) / mean(x), NA_real_)
  )
}) %>%
  mutate(across(where(is.numeric), ~ round(., 3)))

Medidas de forma

Asimetría (coeficiente de Fisher) y curtosis (con exceso respecto a la normal).

asimetria <- function(x) {
  x <- x[!is.na(x)]
  m <- mean(x); s <- sd(x)
  mean((x - m)^3) / s^3
}

curtosis <- function(x) {
  x <- x[!is.na(x)]
  m <- mean(x); s <- sd(x)
  mean((x - m)^4) / s^4
}

tabla_forma <- map_dfr(vars_evaluar, function(v) {
  x <- tabla_base_agrupada[[v]]
  tibble(
    variable        = v,
    asimetria       = asimetria(x),
    curtosis        = curtosis(x),
    exceso_curtosis = curtosis(x) - 3
  )
}) %>%
  mutate(
    across(where(is.numeric), ~ round(., 3)),
    interpretacion_asimetria = case_when(
      asimetria > 0.5  ~ "asimetria positiva (cola derecha)",
      asimetria < -0.5 ~ "asimetria negativa (cola izquierda)",
      TRUE             ~ "aprox. simetrica"
    ),
    interpretacion_curtosis = case_when(
      exceso_curtosis > 0.5  ~ "leptocurtica (colas pesadas)",
      exceso_curtosis < -0.5 ~ "platicurtica (colas livianas)",
      TRUE                   ~ "mesocurtica (aprox. normal)"
    )
  )

Distribución de frecuencias

La tabla agrupada no contiene variables categóricas nativas, por lo que se construye la categoría actividad semanal del cuadrante a partir de n_eventos. Se reportan frecuencias absolutas, relativas y acumuladas.

tabla_frecuencias <- tabla_base_agrupada %>%
  mutate(
    categoria_actividad = case_when(
      n_eventos == 0              ~ "Sin eventos",
      n_eventos == 1              ~ "1 evento",
      n_eventos >= 2 & n_eventos <= 5 ~ "2 a 5 eventos",
      n_eventos > 5               ~ "Mas de 5 eventos"
    ),
    categoria_actividad = factor(
      categoria_actividad,
      levels = c("Sin eventos", "1 evento", "2 a 5 eventos", "Mas de 5 eventos")
    )
  ) %>%
  count(categoria_actividad, name = "frec_absoluta") %>%
  mutate(
    frec_relativa      = round(frec_absoluta / sum(frec_absoluta), 4),
    frec_rel_pct       = round(100 * frec_relativa, 2),
    frec_abs_acumulada = cumsum(frec_absoluta),
    frec_rel_acumulada = round(cumsum(frec_relativa), 4)
  )

Representación gráfica

Todos los gráficos se guardan como PNG en UBICACION_DATA y se copian a la carpeta de imágenes de la tesis si está disponible.

guardar_plot <- function(plot, archivo, width, height) {
  ggsave(
    file.path(UBICACION_DATA, archivo),
    plot = plot,
    width = width,
    height = height,
    dpi = 300,
    bg = "white"
  )
  if (dir.exists(path_img_tesis)) {
    file.copy(
      file.path(UBICACION_DATA, archivo),
      file.path(path_img_tesis, archivo),
      overwrite = TRUE
    )
  }
}
