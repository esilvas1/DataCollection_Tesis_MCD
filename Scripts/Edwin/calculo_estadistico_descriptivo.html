Función de resumen descriptivo

resumen_descriptivo <- function(x, incluir_pct_cero = FALSE) {
  n_total <- length(x)
  x_ok <- x[!is.na(x)]
  n <- length(x_ok)
  n_na <- n_total - n
  pct_na <- if (n_total > 0) round(100 * n_na / n_total, 4) else NA_real_

  if (n == 0) {
    return(tibble(
      n_total = n_total, n = 0L, n_na = n_na, pct_na = pct_na,
      pct_cero = NA_real_, media = NA_real_, sd = NA_real_,
      min = NA_real_, p25 = NA_real_, mediana = NA_real_,
      p75 = NA_real_, max = NA_real_, cv_pct = NA_real_
    ))
  }

  q <- quantile(x_ok, probs = c(0.25, 0.50, 0.75), names = FALSE)
  media_v <- mean(x_ok)
  sd_v <- sd(x_ok)
  pct_cero <- if (incluir_pct_cero) round(100 * mean(x_ok == 0), 4) else NA_real_
  cv_pct <- if (!is.na(media_v) && media_v != 0) round(100 * sd_v / abs(media_v), 4) else NA_real_

  tibble(
    n_total = n_total,
    n = n,
    n_na = n_na,
    pct_na = pct_na,
    pct_cero = pct_cero,
    media = round(media_v, 4),
    sd = round(sd_v, 4),
    min = min(x_ok),
    p25 = q[1],
    mediana = q[2],
    p75 = q[3],
    max = max(x_ok),
    cv_pct = cv_pct
  )
}

Resumen temporal (días con interrupción por año)

eventos_por_anio <- tabla_base_consolidada %>%
  mutate(anio = year(FECHA_DIA)) %>%
  summarise(
    n_filas = n(),
    dias_con_elementos = sum(CANTIDAD_ELEMENTOS > 0, na.rm = TRUE),
    dias_con_usuarios = sum(CANTIDAD_USUARIOS > 0, na.rm = TRUE),
    dias_con_duracion = sum(DURACION_min > 0, na.rm = TRUE),
    dias_con_frecuencia = sum(FRECUENCIA > 0, na.rm = TRUE),
    .by = anio
  ) %>%
  mutate(
    pct_dias_con_evento = round(100 * dias_con_frecuencia / n_filas, 4)
  )
